## ðŸ“„ File: `assets/js/app.js`

```javascript
const { createApp } = Vue
const apiUrl = 'http://localhost/labci4/public' // ganti dengan URL API kamu

createApp({
  data() {
    return {
      artikel: [],
      form: {
        id: null,
        judul: '',
        isi: ''
      }
    }
  },
  mounted() {
    this.loadData()
  },
  methods: {
    loadData() {
      axios.get(apiUrl + '/post')
        .then(response => {
          this.artikel = response.data
        })
        .catch(error => console.error(error))
    },
    submitForm() {
      if (this.form.id) {
        // Edit data
        axios.put(`${apiUrl}/post/${this.form.id}`, this.form)
          .then(() => {
            this.loadData()
            this.resetForm()
          })
          .catch(error => console.error(error))
      } else {
        // Tambah data
        axios.post(apiUrl + '/post', this.form)
          .then(() => {
            this.loadData()
            this.resetForm()
          })
          .catch(error => console.error(error))
      }
    },
    editData(item) {
      this.form = { ...item }
    },
    deleteData(id) {
      if (confirm('Yakin ingin menghapus data ini?')) {
        axios.delete(`${apiUrl}/post/${id}`)
          .then(() => {
            this.loadData()
          })
          .catch(error => console.error(error))
      }
    },
    resetForm() {
      this.form = {
        id: null,
        judul: '',
        isi: ''
      }
    }
  }
}).mount('#app')
